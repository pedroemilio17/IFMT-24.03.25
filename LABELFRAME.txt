import java.awt.FlowLayout;
import javax.swing.Icon;
import javax.swing.ImageIcon;
import javax.swing.JFrame;
import javax.swing.JLabel;
import javax.swing.SwingConstants;

public class LabelFrame extends JFrame{
    private JLabel label1;
    private JLabel label2;
    private JLabel label3;

    public LabelFrame(){
        super("Testando Label");
        setLayout(new FlowLayout());

        label1 = new JLabel("Label com texto");
        label1.setToolTipText("Esta é a label 1");
        add(label1);

        Icon bug = new ImageIcon(getClass().getResource("ifmt2.png"));
        label2 = new JLabel("Label com texto e ícone ", bug, SwingConstants.LEFT);
        label2.setToolTipText("Esta é a label 2");
        add(label2);

        label3 = new JLabel();
        label3.setText("Label com texto e ícone abaixo");
        label3.setIcon(bug);
        label3.setHorizontalTextPosition(SwingConstants.CENTER);
        label3.setVerticalTextPosition(SwingConstants.BOTTOM);
        label3.setToolTipText("Esta é a label 3");
        add(label3);
    }
}
